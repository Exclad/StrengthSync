    Step 3 — Write the Playwright script to `/tmp/pw_phase6.js`:

    ```javascript
    const { chromium } = require('/tmp/node_modules/playwright');
    (async () => {
      const browser = await chromium.launch({
        executablePath: '/root/.cache/ms-playwright/chromium-1217/chrome-linux64/chrome',
        args: ['--no-sandbox']
      });

      const errors = [];
      const page = await browser.newPage();
      page.on('pageerror', e => errors.push(e.message));
      await page.setViewportSize({ width: 1400, height: 900 });

      // --- LIGHT MODE: Sync screen ---
      await page.goto('http://localhost:5000', { waitUntil: 'networkidle' });
      await page.waitForTimeout(4000);
      await page.screenshot({ path: '/tmp/pw_phase6_light.png' });
      console.log('✓ Light mode sync screenshot taken');

      // --- DARK MODE: Brand mark visibility ---
      await page.evaluate(() => {
        document.body.setAttribute('data-theme', 'dark');
      });
      await page.waitForTimeout(500);
      await page.screenshot({ path: '/tmp/pw_phase6_dark.png' });
      console.log('✓ Dark mode brand mark screenshot taken');

      // --- LIBRARY: Empty state ---
      await page.evaluate(() => { document.querySelectorAll('.nav-pill button')[1].click(); });
      await page.waitForTimeout(1500);
      await page.screenshot({ path: '/tmp/pw_phase6_library.png' });
      console.log('✓ Library screenshot taken');

      // --- HISTORY: Empty state ---
      await page.evaluate(() => { document.querySelectorAll('.nav-pill button')[2].click(); });
      await page.waitForTimeout(1500);
      await page.screenshot({ path: '/tmp/pw_phase6_history.png' });
      console.log('✓ History screenshot taken');

      // --- SETTINGS: All 4 groups ---
      await page.evaluate(() => { document.querySelectorAll('.nav-pill button')[3].click(); });
      await page.waitForTimeout(1500);
      await page.screenshot({ path: '/tmp/pw_phase6_settings.png' });
      console.log('✓ Settings screenshot taken');

      // --- SETTINGS LIGHT MODE ---
      await page.evaluate(() => {
        document.body.setAttribute('data-theme', 'light');
      });
      await page.waitForTimeout(300);
      await page.screenshot({ path: '/tmp/pw_phase6_settings_light.png' });
      console.log('✓ Settings light mode screenshot taken');

      // --- Report errors ---
      if (errors.length > 0) {
        console.log('JS ERRORS:', errors);
      } else {
        console.log('✓ No JS errors detected');
      }

      await browser.close();
      console.log('Done. Screenshots at /tmp/pw_phase6_*.png');
    })();
    ```
