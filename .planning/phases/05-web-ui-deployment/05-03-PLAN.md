3. On mount, fetch suggestions for all non-cardio exercises in parallel:
   ```javascript
   useEffect(() => {
     // Fetch GET /api/exercises for search-all fallback
     fetch('/api/exercises').then(r => r.json()).then(setAllGarminExercises).catch(() => {});

     // Fetch suggestions for each unmapped exercise
     setExercises(prev => prev.map(ex => {
       if (ex.status === 'cardio') return ex;
       fetch('/api/map/suggest', {
         method: 'POST',
         headers: { 'Content-Type': 'application/json' },
         body: JSON.stringify({ hevy_exercise_name: ex.hevy }),
       })
         .then(r => r.json())
         .then(body => {
           const suggestions = body.suggestions || [];
           const top = suggestions[0];
           const newStatus = top && top.score >= 70 ? 'mapped' : (top ? 'needs-review' : 'unmapped');

           setExercises(prev2 => prev2.map(e =>
             e.id === ex.id
   ```

4. handleConfirm(exId, suggestion): POST /api/map/confirm, then update exercise status to 'mapped':
   ```javascript
   const handleConfirm = (exId, suggestion) => {
     setMapError(null);
     fetch('/api/map/confirm', {
       method: 'POST',
       headers: { 'Content-Type': 'application/json' },
       body: JSON.stringify({ hevy_name: exercises.find(e => e.id === exId)?.hevy, garmin_name: suggestion.id }),
     })
       .then(async r => {
         if (!r.ok) { const b = await r.json(); setMapError(b.error || 'Confirm failed.'); return; }
         setExercises(prev => prev.map(e =>
           e.id === exId
             ? { ...e, status: 'mapped', confidence: 1.0, garmin: suggestion.id, garminLabel: suggestion.label }
             : e
         ));
         // Auto-advance to next UNRESOLVED exercise
         const unresolvedIds = exercises.filter(e => e.status === 'needs-review' || e.status === 'unmapped').map(e => e.id);
         const nextId = unresolvedIds.find(id => id !== exId);
         if (nextId) setSelectedId(nextId);
       })
       .catch(() => setMapError('Network error.'));
   };
   ```
